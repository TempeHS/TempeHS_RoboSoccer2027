import sensor, image, time
sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)
sensor.skip_frames(time = 2000)

sensor.set_auto_gain(False)
sensor.set_auto_whitebal(False)

orange_threshold = (34, 64, 24, 127, 0, 127)

clock = time.clock()

while(True):
    clock.tick()
    img = sensor.snapshot()

    blobs = img.find_blobs([orange_threshold], pixels_threshold=200, area_threshold=200, merge=True)

    if blobs:
        max_blob = max(blobs, key=lambda b: b.pixels())

        img.draw_rectangle(max_blob.rect(), color=(0, 255, 0), thickness=2)
        img.draw_cross(max_blob.cx(), max_blob.cy(), color=(255, 0, 0), size=10)

        print(f"Tracking Ball -> X: {max_blob.cx()} | Y: {max_blob.cy()} | Width: {max_blob.w()}")
    else:
        print("Ball not detected")
